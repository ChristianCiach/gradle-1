set STARTER_MAIN_CLASS=org.gradle.BootstrapMain

set JAVA_EXE=%JAVA_HOME%\bin\java.exe
set TOOLS_JAR=%JAVA_HOME%\lib\tools.jar

if "%JAVA_OPTS%" == "" set JAVA_OPTS="-Xmx128m"
set JAVA_OPTS=%JAVA_OPTS% -Dgradle.home="%GRADLE_HOME%"
set JAVA_OPTS=%JAVA_OPTS% -Dtools.jar="%TOOLS_JAR%"

@rem Execute Gradle
"%JAVA_EXE%" %JAVA_OPTS% -classpath "%CLASSPATH%" %STARTER_MAIN_CLASS% %CMD_LINE_ARGS%

:end
@rem End local scope for the variables with windows NT shell
if "%ERRORLEVEL%"=="0" goto mainEnd

if "%OS%"=="Windows_NT" color 00
if not "%OS%"=="Windows_NT" echo 1 > nul | choice /n /c:1
goto omega

:mainEnd
if "%OS%"=="Windows_NT" endlocal

:omega
